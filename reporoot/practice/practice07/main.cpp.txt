#include <iostream>
#include <vector>

class Employee {
protected:
    std::string name;
    int id;

public:
    Employee(std::string name, int id) : name(name), id(id) {}
    virtual ~Employee() {}

    virtual double calculateSalary() const = 0; // Pure virtual function
    virtual void displayInfo() const {
        std::cout << "ID: " << id << ", Name: " << name;
    }
};
